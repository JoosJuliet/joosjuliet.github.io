문제:
https://www.acmicpc.net/problem/1212

``` python
import sys
read = lambda : sys.stdin.readline().strip()

h = read()

ans = ''
for i in h:
    tmp_ans = ''
    i = int(i)
    while i > 0:
        tmp_ans += str(i % 2)
        i //= 2
    tmp_ans = tmp_ans[::-1]
    if len(tmp_ans) <3:
        tmp_ans = '0'*(3-len(tmp_ans)) + tmp_ans
    ans += tmp_ans
tmp_i = -1
for i in range(len(ans)):
    if ans[i] == '1':
        tmp_i = i
        break

if tmp_i == -1:
    print(0)
else:
    print(''.join(ans[tmp_i:]))
```
