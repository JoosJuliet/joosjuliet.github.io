문제: https://www.acmicpc.net/problem/2447

``` JAVA

import java.util.*;

public class Main {
    private static final Scanner sc = new Scanner(System.in);
    private static char[][] starMap;

    public static void main(String args[]) {
        int num = sc.nextInt();
        starMap = new char[num][num];

        solve(0, 0, num, false);
        System.out.println(' ');
        for (int i = 0; i < num; i++) {
            System.out.println(Arrays.toString(starMap[i]));
        }
    }

    public static void solve (int x, int y, int n, boolean flag) {
        if (flag == true){
            for (int i = x; i < x+n; i++) {
                for (int j = y; j < y+n; j++) {
                    starMap[i][j] = ' ';
                }
            }
        } else {
            if (n == 1){
                starMap[x][y] = '*';
                return;
            }
            int m = n/3;
            for (int i=0; i<3; i++) {
                for (int j=0; j<3; j++) {
                    if (i==1 && j==1){
                        flag = true;
                    } else{
                        flag = false;
                    }
                    solve(x+m*i, y+m*j, m, flag);
                }
            }


        }
    }

}
```
