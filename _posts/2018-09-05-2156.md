``` python
n = int(input())
glasses = [0] * (n+1)
answer = [0] * (n+1)
for i in range(1,n+1):
    glasses[i] = int(input())

answer[1] = glasses[1]
if n >= 2:
    answer[2] = answer[1]+glasses[2]

for i in range(3,n+1):
    answer[i] = answer[i-1]
    answer[i] = max(answer[i],answer[i-2]+glasses[i])
    answer[i] = max(answer[i],answer[i-3]+glasses[i-1]+glasses[i])

print(max(answer))
```
