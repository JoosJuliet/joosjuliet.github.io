# dfs
``` python
import sys
read = lambda : sys.stdin.readline().strip()
sys.setrecursionlimit(100000)
dx = [0, 0, -1, 1, -1, -1, 1, 1]
dy = [1, -1, 0, 0, 1, -1, -1, 1]

def dfs(matrix, x, y):
    matrix[x][y] = 0
    for k in range(8):
        nx = x + dx[k]
        ny = y + dy[k]
        if 0 <= nx and nx < h and 0 <= ny and ny < w:
            if matrix[nx][ny] == 1:
                # matrix[nx][ny] = 0
                dfs(matrix, nx, ny)
while True:
    w, h = map(int, read().split())
    if w == h == 0 : break
    matrix = []
    for _ in range(h):
        matrix.append(list(map(int, read().split())))
    cnt = 0
    for i in range(h):
        for j in range(w):
            if matrix[i][j] == 1:
                dfs(matrix, i, j)
                cnt += 1
    print(cnt)
```
