``` python

import sys
t = int(sys.stdin.readline())

d = [0,1,1,1,2,2,3,4,5,7,9]
def solving(n):
    for _ in range(n):
        d.append(0)
    if d[n] > 0:
        return str(d[n])
    ans = 0
    k = 9
    for i in range(10,n+1):
        d[i] = d[i-1]+d[i-5]
    return str(d[n])

for _ in range(t):
    sys.stdout.write(solving(int(sys.stdin.readline())))
    sys.stdout.write('\n')
```
