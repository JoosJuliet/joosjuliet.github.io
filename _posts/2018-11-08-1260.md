문제: https://www.acmicpc.net/problem/1260

``` python
n, m, v = map(int, input().split())

vertexs = {}
for i in range(n):
    vertexs[i+1] = []
for _ in range(m):
    a, b = map(int, input().split())
    vertexs[a].append(b)
    vertexs[b].append(a)
check = [False]*(n+1)

def dfs(n):
    print(n,end=' ')
    check[n] = True
    for k in sorted(vertexs[n]):
        if check[k] == False:
            dfs(k)

dfs(v)

print()
check = [False]*(n+1)
queue = [v]
check[v] = True
print(v,end=' ')
while queue:
    v = queue.pop(0)
    for u in sorted(vertexs[v]):
        if check[u] == True : continue
        print(u,end=' ')
        queue.append(u)
        check[u] = True

```
