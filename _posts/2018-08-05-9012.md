```python3
N = int(input())

def solving():
    statement= input()
    stack = []
    for i in statement:
        if i == '(':
            stack.append(i)
        else:
            if len(stack) == 0:
                return 'NO'
            else:
                stack.pop()
    if len(stack) > 0:
        return 'NO'
    else:
        return 'YES'
while(N):
    print(solving())
    N -= 1
```

#made by LaVi
N = int(input())

def solving():

    statement= input()

    left = 0
    right = 0
    for i in statement:
        if i == '(':
            left += 1
        else:
            right += 1
        if left < right:
            break
    if left != 0 and right != 0 and left == right:
        return 'YES'
    else:
        return 'NO'

while(N):
    print(solving())
    N -= 1
