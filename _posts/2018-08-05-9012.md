```python3
N = int(input())

def solving():
    statement= input()
    stack = []
    for i in statement:
        if i == '(':
            stack.append(i)
        else:
            if len(stack) == 0:
                return 'NO'
            else:
                stack.pop()
    if len(stack) > 0:
        return 'NO'
    else:
        return 'YES'
while(N):
    print(solving())
    N -= 1
```
