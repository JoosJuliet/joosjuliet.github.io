# 방법 1

``` python

import sys
read = lambda : sys.stdin.readline().strip()
sys.setrecursionlimit(100000)

def dfs(graph, color, x, c):
    color[x] = c
    for y in graph[x]:
        if color[y] == 0:
            dfs(graph, color, y, 3-c)


def make_graph(v, e):
    graph = {}
    for i in range(1,v+1):
        graph[i] = set()
    for j in range(e):
        a, b = map(int, read().split())
        graph[a].add(b)
        graph[b].add(a)
    return graph


def solution():
    v, e = map(int, read().split())
    graph = make_graph(v, e)
    color = [0]*(v+1)
    ok = 'YES'

    for i in range(1, v+1):
        if color[i] == 0:
            dfs(graph, color, i, 1)
            # 1로 일단 무조건 하고 dfs속에서는 2하나만 하게하고 끝내는 건가

    for i in range(1, v+1):
        for j in graph[i]:
            if color[i] == color[j]:
                ok = 'NO'

    print(ok)

for i in range(int(read())):
    solution()

```

import sys
read = lambda : sys.stdin.readline().strip()
sys.setrecursionlimit(100000)


def dfs(start, c):
	check[start] = c
	for node in graph[start]:
		if check[node]==0:
			if dfs(node, 3-c) == False:
				return False
		else:
			if check[node]==check[start]:
				return False
	return True


for _ in range(int(read())):
    v, e = map(int, read().split())
    graph = {}
    for i in range(1,v+1):
        graph[i] = set()
    for j in range(e):
        a, b = map(int, read().split())
        graph[a].add(b)
        graph[b].add(a)

    check = [0]*(v+1)
    ok = True
    for i in range(v):
    	if check[i+1]==0:
    		if dfs(i+1, 1)==False:
    			ok = False
    			break
    print("YES" if ok else "NO")
